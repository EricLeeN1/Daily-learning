$(document).ready(function() {

    var badge = 5;

    var favicon = new Favico({

        animation : 'popFade'

    });

    $('#plusOne').bind('click',function() {

        badge = badge + 1;

        favicon.badge(badge);

    });

    $('#minusOne').bind('click',function() {

        badge = (badge-1 < 0) ? 0 : (badge - 1);

        favicon.badge(badge);

    });

    $('#reset').bind('click',function() {

        favicon.reset();

    });

    //初始值

    favicon.badge(badge);

});
